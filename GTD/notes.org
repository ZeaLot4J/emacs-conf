#+BEGIN_SRC java
public class Singleton {
    private final static Singleton INSTANCE = new Singleton();
    private Singleton() {
        if (INSTANCE != null) {
	    throw RuntimeException("There can be only one instance of " + Singleton.class);
        }
    }
    public static Singleton getInstance(){
        return INSTANCE;
    }
}
